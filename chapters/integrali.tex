\chapter{integrali multipli}

\section{integrale di Riemann}
Se $x,y\in \RR$ definiamo:
\[
    x \vee y \defeq \max\ENCLOSE{x,y},
    \quad
    x \wedge y \defeq \min\ENCLOSE{x,y},
    \quad
    x^+ \defeq x \vee 0,
    \quad
    x^- \defeq (-x)^+ \defeq -(x\wedge 0).
\]
Si ha
\[
    x^+,x^-\ge 0,
    \quad
    x = x^+ - x^-,
    \quad 
    \abs{x} = x^+ + x^-,
    \quad
    x^+ = \frac{\abs{x}+x}{2},
    \quad
    x^- = \frac{\abs{x}-x}{2}.
\]

Sia $A\subset \RR^n$ e $f\colon A \to \RR$. 
Definiamo il \emph{supporto} di $f$ come 
\[
 \spt f 
    \defeq \overline{\ENCLOSE{\vec x \in A \colon f(\vec x) \neq 0}}
    = \overline{f^{-1}(\RR\setminus\ENCLOSE{0})}.
\]

Dato $A\subset \RR^n$ definiamo la \emph{funzione caratteristica}
di $A$ come $1_A\colon \RR^n \to \RR$,
\[
  1_A(\vec x) \defeq \begin{cases}
    1 & \text{se $\vec x \in A$,}\\
    0 & \text{altrimenti.}
  \end{cases}
\]

Per $\vec k \in \ZZ^n$ e $N\in \NN$ definiamo 
il cubetto 
\[
  Q^N_{\vec k} \defeq \frac{\vec k + [0,1]^d}{2^N}.
\]
Si tratta di un cubetto di lato $2^{-N}$ con gli spigoli paralleli 
agli assi cartesiani e con un vertice nel punto $2^{-N} \vec k$.
Il volume di $Q^N_{\vec k}$ sarﺣ  quindi $2^{-nN}$.
Useremo questi cubetti per misurare il volume di insiemi arbitrari.

\begin{definition}[integrale di Riemann]
  \label{def:integrale_riemann}
Sia $f\colon \RR^n \to \RR$ una funzione limitata, 
con supporto limitato.
Definiamo le somme superiori e inferiori sulla $N$-esima suddivisione in cubi:
\begin{align*}
    S^N(f) &\defeq 2^{-nN}\sum_{\vec k\in \ZZ^n} \sup f(Q^N_{\vec k}), \\
    S_N(f) &\defeq 2^{-nN}\sum_{\vec k\in \ZZ^n} \inf f(Q^N_{\vec k}).
\end{align*}
Visto che il supporto di $f$ ﺣ۷ limitato, queste somme sono in realtﺣ  
somme finite. Visto che $f$ ﺣ۷ limitata, ogni addendo ﺣ۷ finito e dunque 
$S^N(f), S_N(f) \in \RR$ per ogni $N\in \NN$.
Osservando che $N\mapsto S^N(f)$ ﺣ۷ decrescente 
mentre $N\mapsto S_N(f)$ ﺣ۷ crescente, definiamo
l'integrale superiore e inferiore:
\begin{align*}
    S^*(f) &\defeq \inf_N S^N(f) = \lim_{N\to +\infty} S^N(f) \in \RR, \\
    S_*(f) &\defeq \sup_N S_N(f) = \lim_{N\to +\infty} S_N(f) \in \RR.
\end{align*}
Se $S^*(f)=S_*(f)$ diremo che $f$ ﺣ۷ Riemann integrabile (o R-integrabile)
e scriveremo
\[
  \int\!\! f = S^*(f) = S_*(f).
\]

Volendo rendere esplicita la variabile di integrazione, potremo
usare la notazione equivalente
\[
  \int\!\! f(x)\, dx \defeq \int\!\! f.
\]
\end{definition}

\begin{theorem}[monotonia dell'integrale]
Siano $f,g\colon \RR^n\to\RR$ due funzioni limitate con supporto limitato.
Se $f$ e $g$ sono $R$-integrabili si ha:
\[
  f \le g \implies \int\!\! f \le \int\!\! g.
\]
\end{theorem}
%
\begin{proof}
Se $f\le g$, per ogni $N\in \NN$ e ogni $\vec k \in \ZZ^n$ si ha 
$\sup f(Q^N_{\vec k}) \le \sup g(Q^N_{\vec k})$
da cui 
$S^N(f) \le S^N(g)$.
Passando al limite per $N\to +\infty$ si ottiene
$S^*(f) \le S^*(g)$
che ﺣ۷ la tesi se $f$ e $g$ sono R-integrabili.
\end{proof}

\begin{theorem}[linearitﺣ  dell'integrale]
Siano $f,g\colon \RR^n \to \RR$ funzioni limitate con supporto limitato.
Se $f,g$ sono R-integrabili allora $f+g$ ﺣ۷ R-integrabile e si ha 
\[
    \int (f+g) = \int\!\! f + \int\!\! g.
\]
Se $f$ ﺣ۷ R-integrabile e $c\in \RR$ ﺣ۷ costante, anche $c f$ ﺣ۷ R-integrabile 
e si ha 
\[
    \int\!\! c f = c \!\int\!\! f.
\]
\end{theorem}
\begin{proof}
Per ogni $\vec k\in\ZZ^n$ e ogni $N\in \NN$
si ha
\begin{align*}
  \sup\, (f+g)(Q^N_{\vec k}) &\le \sup f(Q^N_{\vec k}) + \sup g(Q^N_{\vec k})\\
  \inf\, (f+g)(Q^N_{\vec k}) &\ge \inf f(Q^N_{\vec k}) + \inf g(Q^N_{\vec k}).
\end{align*}
Dunque si ottiene
\[
    S^N(f+g) \le S^N(f) + S^N(g),  \qquad
    S_N(f+g) \ge S_N(f) + S_N(g), 
\]
e passando al limite per $N\to +\infty$ 
\[
   S_*(f) + S_*(g) \le S_*(f+g) \le S^*(f+g) \le S^*(f) + S^*(g).
\]
Se $f$ e $g$ sono R-integrabili si ottiene dunque che anche $f+g$ 
ﺣ۷ R-integrabile e inoltre $\int (f+g) = \int\! f + \int\! g$.

Se $c \ge 0$ si ha 
\[
   S^N(c f) = c S^N(f), \qquad S_N(c f) = c S_N(f)
\]
e quindi se $f$ ﺣ۷ R-integrabile si conclude 
che $c f$ ﺣ۷ R-integrabile e $\int\! c f = c \int\! f$.

Se $c = -1$ si osserva che per ogni $N\in \NN$, $\vec k\in \ZZ^n$, 
si ha 
\[
   \sup_{Q^N_{\vec k}} (-f) = -\inf_{Q^N_{\vec k}} f,
   \qquad 
   \inf_{Q^N_{\vec k}} (-f) = -\sup_{Q^N_{\vec k}} f,
\]
da cui 
\[
    S^N(-f) = -S_N(f), \qquad S_N(-f) = -S^N(f)
\]
e quindi, per $N\to +\infty$ si ottiene 
\[
  S^*(-f) = -S_*(f), \qquad S_*(-f) = - S^*(f).
\]
Se $f$ ﺣ۷ R-integrabile deduciamo che anche $-f$ lo ﺣ۷ e si ha 
$\int (-f) = -\int\! f$.

Combinando i casi precedenti si ottiene il caso generale $c \in \RR$.
\end{proof}

\begin{theorem}[proprietﺣ  di reticolo]
Sia $f\colon \RR^n \to \RR$ una funzione limitata con supporto limitato.
Se $f$ ﺣ۷ R-integrabile allora anche $\abs{f}$, $f^+$ e $f^-$ sono 
R-integrabili e si ha 
\[
  \abs{\int\!\! f} \le \int\! \abs f = \int\!\! f^+ + \int\!\! f^-, 
  \qquad 
 \int\!\! f = \int\!\! f^+ - \int\!\! f^-.
\]
\end{theorem}
\begin{proof}
Dalla proprietﺣ  generale 
\[
  \abs{\rule{0pt}{1.1em}\abs{f(\vec x)} - \abs{f(\vec y)}} \le \abs{f(\vec x) - f(\vec y)}
\]
si ottiene che per ogni $N\in \NN$, $\vec k \in \ZZ^n$, vale
\[
  \sup_{Q^N_{\vec k}} \abs{f} - \inf_{Q^N_{\vec k}} \abs{f}
  \le 
  \sup_{Q^N_{\vec k}} f - \inf_{Q^N_{\vec k}} f
\]
e quindi 
\[
   S^N(\abs f) - S_N(\abs f) \le S^N(f) - S_N(f)
\] 
da cui, passando al limite per $N\to +\infty$,
\[
  S^*(\abs f) - S_*(\abs f) \le S^*(f) - S_*(f)
\]
per cui se $f$ ﺣ۷ R-integrabile anche $\abs{f}$ lo ﺣ۷.

Osserviamo ora che $f^+=\frac{\abs f + f}{2}$ e $f^-=\frac{\abs f - f}{2}$ 
e dunque per linearitﺣ  se $f$ ﺣ۷ R-integrabile avendo giﺣ  dimostrato 
che anche $\abs f$ ﺣ۷ R-integrabile, si ottiene che pure $f^+
$ e $f^-$ sono R-integrabili ed essendo $f=f^+-f^-$, $\abs f = f^+ + f^-$ si ottiene 
\[
  \int\!\! f = \int\!\! f^+ - \int\!\! f^-, \qquad 
  \int\! \abs f = \int\!\! f^+ + \int\!\! f^-.
\]
Per la convessitﺣ  del valore assoluto si ottiene infine
\[
 \abs{\int\!\! f} = \abs{\int\!\! f^+ - \int\!\! f^-} 
 \le \abs {\int\!\! f^+} + \abs{\int\!\! f^-}
 = \int\!\! f^+ + \int\!\! f^- = \int\! \abs f.
\]
\end{proof}

\begin{definition}[integrale di Riemann generalizzato]
Se $f\colon \RR^n \to [0,+\infty]$ ﺣ۷ una funzione (non negativa), per ogni $R>0$ 
possiamo definire il troncamento $f_R$ 
\[
  f_R = (f \wedge R) \cdot 1_{[-R,R]^n}.
\]
La funzione $f_R$ ﺣ۷ limitata ed ha supporto limitato.
Se $f_R$ ﺣ۷ Riemann-integrabile per ogni $R>0$ diremo che $f$ ﺣ۷ localmente R-integrabile 
e visto che $R\mapsto f_R(\vec x)$ ﺣ۷ crescente possiamo definire
\[
 \int\!\! f \defeq \lim_{R\to+\infty} \int\!\! f_R \in [0,+\infty].
\]
Se $\int\!\! f<+\infty$ diremo che $f$ ﺣ۷ R-integrabile (avendo supposto $f\ge 0$)

Se $f\colon \RR^n\to \RR$ possiamo scrivere $f= f^+-f^-$ con $f^+,f^-$ funzioni non negative.
Se $f^+$ e $f^-$ sono localmente R-integrabili e se 
$\int\!\! f^+$ e $\int\!\! f^-$ non sono entrambi infiniti possiamo definire 
\[
  \int\!\! f = \int\!\! f^+ - \int\!\! f^- \in [-\infty,+\infty].
\]
Se $\int\!\! f$ ﺣ۷ finito diremo che $f$ ﺣ۷ $R$-integrabile.
\end{definition}

\begin{definition}
Dato $E\subset \RR^n$ potremo definire la misura (di Peano-Jordan) esterna 
e interna di $E$ come 
\begin{align*}
 m^N(E) &\defeq S^N(1_E), & m^*(E) &\defeq S^*(1_E),\\
 m_N(E) &\defeq S_N(1_E), & m_*(E) &\defeq S_*(1_E).
\end{align*}

Diremo che $E$ ﺣ۷ \emph{Peano-Jordan misurabile} 
(PJ-misurabile) se la funzione caratteristica $1_E$ ﺣ۷ localmente 
R-integrabile,
ovvero se $m^*(E\cap[-R,R]^n) = m_*(E\cap[-R,R]^n)$ per ogni $R>0$.
In tal caso definiamo la misura di $E$ come 
\[
\abs{E} \defeq \int\!\! 1_E \in [0,+\infty] = \lim_{R\to+\infty} \abs{E\cap[-R,R]^n}.
\]


Se $A\subset \RR^n$ ﺣ۷ PJ-misurabile e $f\colon A \to \RR$
ﺣ۷ una funzione qualunque, possiamo considerare l'estensione 
$f_A\colon \RR^n \to \RR$ definita da 
\[
 f_A(\vec x) = \begin{cases}
    f(\vec x) & \text{se $x\in A$,}\\
    0 & \text{altrimenti.}
 \end{cases}
\]
Diremo che $f$ ﺣ۷ localmente R-integrabile in $A$ se $f_A$ 
ﺣ۷ localmente R-integrabile. 
Scriveremo:
\[
 \int_A f = \int\!\! f_A
\]
se $\int f_A$ ﺣ۷ definito.
Se inoltre $\int_A f$ ﺣ۷ finito diremo che $f$ ﺣ۷ R-integrabile
su $A$.

Volendo rendere esplicita la variabile di integrazione, potremo 
scrivere 
\[
  \int_A f(x)\, dx \defeq \int_A f.
\]
\end{definition}

\begin{remark}
Visto che $\sup_{Q^N_{\vec k}} 1_E = 1$ se $Q^N_{\vec k}$ tocca $E$ e $0$ altrimenti,
mentre $\inf_{Q^N_{\vec k}} 1_E = 1$ se $Q^N_{\vec k}$ ﺣ۷ contenuto in $E$
e $0$ altrimenti,
si osserva che:
\begin{align*}
 m^N(E) &= 2^{-nN} \cdot \# \ENCLOSE{\vec k\in \ZZ^N\colon Q_{\vec k}^N \cap E \neq \emptyset},\\
  m_N(E) &= 2^{-nN} \cdot \# \ENCLOSE{\vec k\in \ZZ^N\colon Q_{\vec k}^N \subset E}
\end{align*}
e dunque $m^N(E)$ ﺣ۷ il volume dell'unione dei cubetti $Q^N_{\vec k}$ che ricoprono $E$ 
mentre $m_N(E)$ ﺣ۷ il volume dell'unione dei cubetti $Q^N_{\vec k}$ che sono contenuti in $E$.
\end{remark}

\begin{remark}[caso $n=1$]
Nel caso $n=1$ se $f\colon [a,b]\to \RR$ ﺣ۷ una funzione limitata,
si osserva che la suddivisione in cubetti di lato $2^{-N}$ induce una partizione 
su $[a,b]$ e la definizione di Riemann-integrabilitﺣ  data sopra coincide con la 
usuale Riemann-integrabilitﺣ  definita tramite suddivisioni arbitrarie.

Nel caso $f\colon(a,b)\to \RR$ con $(a,b)$ non limitato oppure $f$ non limitata,
la definizione data sopra corrisponde invece alla \emph{assoluta convergenza} 
dell'integrale improprio.
\end{remark}

\begin{exercise}[misurabilitﺣ  dei rettangoli cartesiani]
  \label{ex:misura_rettangolo}
Dati $\vec a,\vec b\in \RR^n$ con $a_k\le b_k$ per ogni $k=1,\ldots,n$,
il rettangolo cartesiano
\[
  R(\vec a, \vec b) = [a_1,b_1]\times \dots \times [a_n,b_n]
\]  
ﺣ۷ PJ-misurabile e si ha 
\[
    \abs{R(\vec a, \vec b)} = \prod_{k=1}^n (b_k - a_k).
\]

In particolare ogni cubetto $Q^N_{\vec k}$ ﺣ۷ PJ-misurabile e si ha 
$\abs{Q^N_{\vec k}} = 2^{-nN}$.
\end{exercise}
\begin{proof}
Osserviamo che
\[
  Q^N_{\vec k} = [2^{-N} k_1, 2^{-N}(k_1+1)] \times \dots \times [2^{-N} k_n, 2^{-N}(k_n+1)]
\]
ﺣ۷ un rettangolo cartesiano.
Fissato $N$ si ha $Q^N_{\vec k} \subset R(\vec a,\vec b)$
per tutti i $\vec k = (k_1,\dots,k_n) \in \ZZ^n$ tali
che
\[
  2^N a_j \le k_j \le 2^N b_j + 1 \qquad \text{per ogni } j=1,\ldots,n.  
\]
Dunque il numero di cubetti $Q^N_{\vec k}$ contenuti in $R(\vec a,\vec b)$ 
ﺣ۷ non inferiore a 
\[
  \prod_{j=1}^n (2^N(b_j-a_j)-3)
\]
e quindi 
\[
  m_N(R(\vec a,\vec b)) 
    \ge 2^{-nN} \prod_{j=1}^n (2^N(b_j-a_j)-3)
    = \prod_{j=1}^n (b_j - a_j - 3\cdot 2^{-N})
    \to \prod_{j=1}^n (b_j - a_j) \quad \text{per } N\to +\infty.
\]
In maniera analoga si dimostra che 
\[
  m^N(R(\vec a,\vec b)) 
  \le \prod_{j=1}^n (b_j - a_j + 3\cdot 2^{-N})
  \to \prod_{j=1}^n (b_j - a_j) \quad \text{per } N\to +\infty.
\]
Si ottiene dunque il risultato voluto.
\end{proof}

\begin{theorem}[algebra dei misurabili]
Se $E,F\subset \RR^n$ sono PJ-misurabili allora anche $\bar E$, $\partial E$, 
$E\setminus F$, $E\cap F$ ed $E\cup F$ sono PJ-misurabili.
Inoltre si ha 
\[
    \abs{E\cup F} + \abs{E\cap F} = \abs{E} + \abs{F},
    \qquad
    \abs{\partial E} = 0.
\]
\end{theorem}
\begin{proof}
Per ottenere la misurabilitﺣ  di $E\setminus F$ e di $E\cap F$, basta osservare che 
\begin{align*}
  1_{E\setminus F} &= (1_E - 1_F)^+\\
  1_{E\cap F} &= 1_E - 1_{E\setminus F}\\
  1_{E\cup F} &= 1_F + 1_{E\setminus F}
\end{align*}
e usare le proprietﺣ  di reticolo.
Osservando poi che 
\[
  1_{E\cup F} + 1_{E\cap F} = 1_E + 1_F 
\]
si ottiene la formula
\[
  \abs{E\cup F} + \abs{E\cap F} = \abs{E} + \abs{F}.
\]

Supponiamo ora che $E$ sia limitato
e dimostriamo che $\partial E$ ha misura nulla.
Bisogna osservare 
che $\sup 1_{\partial E}(Q^N_{\vec k}) = 1$ 
significa che $\partial E\cap Q^N_{\vec k} \neq \emptyset$.
Ma allora c'ﺣ۷ un punto $\vec x\in Q^N_{\vec k}$ tale che ogni intorno 
di $\vec x$ contiene sia punti di $E$ che punti di $\RR^n\setminus E$.
Se il punto $\vec x$ ﺣ۷ interno a $Q^N_{\vec k}$ si ha
\[
  \sup 1_{\partial E}(Q^N_{\vec k}) > \inf 1_{\partial E}(Q^N_{\vec k}).
\]
Se ciﺣﺎ non fosse vero significa che 
$Q^N_{\vec k}$ ﺣ۷ interamente contenuto in $E$ o in $\RR^n\setminus E$
e che $\vec x$ sta sulla frontiera del cubetto.
Ma allora la disuguaglianza precedente risulta valida 
per un cubetto $Q^N_{\vec k'}$ adiacente a $Q^N_{\vec k}$.
Dunque si deve avere 
\[
  m^N(\partial E)\le 3^n (m^N(E)-m_N(E))
\]
(in quanto $3^n-1$ ﺣ۷ il numero di cubetti adiacenti ad un cubetto fissato)
e per $N\to +\infty$ si ottiene 
\[
  m^*(\partial E) \le 3^n (m^*(E)-m_*(E)) = 0, 
  \qquad 
  m_*(\partial E)\ge 0.
\]
Dunque $\partial E$ ﺣ۷ misurabile e $\abs{\partial E} = 0$.

Se $E$ non ﺣ۷ limitato si ripete l'osservazione precedente alle intersezioni 
$E\cap [-R,R]^n$ per ottenere lo stesso risultato.

Visto che $\bar A = A \cup \partial A$, se $A$ ﺣ۷ misurabile anche $\bar A$ ﺣ۷ 
misurabile e si ha $\abs{\bar A} = \abs{A\setminus \partial A} + \abs{\partial A}
= \abs{A \setminus \partial A} \le \abs{A} \le \abs{\bar A}$,
da cui $\abs{\bar A}=\abs{A}$.
\end{proof}

\begin{theorem}[significato geometrico dell'integrale]
Se $f\colon A\subset \RR^n \to \RR$ definiamo la parte positiva dell'epigrafico di $f$
come:
\[
  E_f = \ENCLOSE{(\vec x,y)\in A\times \RR \colon 0 \le y \le f(\vec x)} \subset \RR^{n+1}.
\]
Se $A\subset \RR^n$ ﺣ۷ PJ-misurabile ed $f$ ﺣ۷ localmente integrabile 
allora $E_{f^+}$ ed $E_{f^-}$ sono localmente misurabili e, se non
hanno entrambi misura infinita, si ha 
\[
  \int_A f = \abs{E_{f^+}} - \abs{E_{f^-}}.
\]
\end{theorem}

\begin{proof}
Supponiamo $f\ge 0$, $f$ limitata, $A$ limitato e PJ-misurabile.
In tal caso si puﺣﺎ osservare che 
per ogni cubetto $n$-dimensionale che tocca la base $A$, 
i cubetti $(n+1)$-dimensionali che coprono il grafico di $f$ 
sono al massimo due in piﺣﺗ rispetto all'altezza della colonna, 
ovvero del $\sup$ di $f$ sul cubetto base. 
Per i cubetti interni si ha invece l'opposto. Cioﺣ۷:
\begin{align*}
    0 & \le m^N(E_{f^+}) - S^N(f) \le 2\cdot 2^{-N} m^N(A),\\
    0 & \le S_N(f) - m_N(E_{f^+}) \le 2\cdot 2^{-N} m^N(A)
\end{align*}
e, per $N\to +\infty$, si ottiene 
\begin{align*}
  0 &\le m^*(E_{f^+}) - S^*(f) \le 0\cdot m^*(A), \\
  0 &\le S_*(f) - m_*(E_{f^+}) \le 0\cdot m^*(A), 
\end{align*}
ovvero $\abs{E_{f^+}}=\int\! f$, come volevamo dimostrare.

Gli altri casi vengono di conseguenza separando parte positiva e parte negativa 
e passando al limite nei troncamenti.
\end{proof}

\begin{theorem}[integrabilitﺣ  funzioni continue]
Sia $A\subset \RR^n$ un insieme chiuso PJ-misurabile e $f\colon A \to \RR$ una funzione 
continua. Allora $f$ ﺣ۷ localmente R-integrabile. 
Se inoltre $\abs{A}<+\infty$ e $f$ 
ﺣ۷ limitata, allora $f$ ﺣ۷ R-integrabile.
\end{theorem}
%
\begin{proof}
\emph{Passo 1: supponiamo $A$ compatto PJ-misurabile, 
$f$ continua, limitata, non negativa.}
Per il teorema di Weierstrass esiste $M>0$ tale che $0\le f(x) \le M$ per 
ogni $\vec x \in A$.
Fissiamo $\eps>0$.
Per il teorema di Heine-Cantor esiste $\delta>0$ tale 
che se $\vec x_1,\vec x_2\in A$ e $\abs{\vec x_1 -\vec x_2}<\delta$ 
allora $\abs{f(\vec x_1)-f(\vec x_2)}< \eps$. 
Scegliamo $N\in \NN$ abbastanza grande in modo che si abbia $2^{-N}\sqrt n < \delta$
e in modo che $m^N(\partial A)<\eps$.
Allora si ha
\[
  S^N(f) - S_N(f) 
  = 2^{-nN}\sum_{\vec k} (\max f(Q^N_{\vec k})-\min f(Q^N_{\vec k})) 
  \le 2^{-nN}\eps m^N(A).
\]
Visto che $f$ ﺣ۷ limitata e $A$ ﺣ۷ limitato, $S^N(f)$ ﺣ۷ limitata in $N$ 
e facendo tendere $N\to+\infty$ si ottiene 
dunque $S^*(f)-S_*(f) \le 0$ da cui $S^*(f)=S_*(f)$ come volevamo dimostrare.

\emph{Passo 2: supponiamo $A$ chiuso PJ-misurabile, $f$ continua, non negativa.} 
Si applica il passo precedente alla troncata $f_R$.

\emph{Passo 3: caso generale.}
Si applica il passo precedente alle parti 
positiva e negativa $f^+$, $f^-$ di $f$.

\emph{Passo 4: supponiamo $\abs{A}<+\infty$ e $f$ limitata.}  
In tal caso esiste $M$ tale che $\abs{f(\vec x)}\le M$ per 
ogni $x\in A$. 
Per la monotonia dell'integrale si ottiene 
\[
  \abs{\int_A f} 
  \le \int_A \abs{f}
  \le \int_A M = M \cdot \abs{A} < +\infty.  
\]
\end{proof}

\begin{theorem}[Fubini-Tonelli]
Sia $f\colon \RR^n\times \RR^m \to \RR$ Riemann integrabile
tale che per ogni $\vec x\in \RR^n$ la funzione $\vec y \mapsto f(\vec x,\vec y)$ 
risulti anch'essa R-integrabile su $\RR^m$.
Allora 
\[
  \int_{\RR^{n+m}} f = \int_{\RR^n} \Enclose{\int_{\RR^m} f(\vec x, \vec y)\, d\vec y}\, d\vec x.
\]
\end{theorem}

\begin{proof}
Supponiamo che $f$ sia una funzione R-integrabile, 
limitata con supporto limitato.
Allora posto 
\[
    g(\vec x) = \int\!\! f(\vec x, \vec y)\, d\vec y\\
\]
si ha
\begin{align*}
    S^N(g)
    &= 2^{-nN} \sum_{\vec k\in \ZZ^n} 
        \sup_{\vec x\in Q^N_{\vec k}}
            \int\!\! f(\vec x,\vec y)\, dy\\
    &\le 2^{-nN} \sum_{\vec k\in \ZZ^n} 
        \sup_{\vec x\in Q^N_{\vec k}}
                S^N(f(\vec x,\cdot))\\
    &\le 2^{-nN} \sum_{\vec k\in \ZZ^n} 
        \sup_{\vec x\in Q^N_{\vec k}}
            2^{-mN} \sum_{\vec j \in \ZZ^m} 
                \sup_{\vec y \in Q^N_{\vec j}} 
                    f(\vec x, \vec y) \\
    &\le 2^{-(n+m)N} \sum_{\vec k\in \ZZ^n} 
        \sum_{\vec j \in \ZZ^m} 
        \sup f(Q^N_{\vec k}\times Q^N_{\vec j})\\
    &= S^N(f)
\end{align*}
ed analogamente si trova $S_N(g) \ge S_N(f)$.

Passando al limite otteniamo quindi 
\[
   S_*(f) \le S_*(g) \le S^*(g) \le S^*(f)
\]
e dunque se $f$ ﺣ۷ integrabile (in $n+m$ variabili) anche $g$ ﺣ۷ integrabile
(in $n$ variabili) 
e risulta $\int f = \int g$, come volevamo dimostrare.

Gli altri casi si ottengono di conseguenza.
\end{proof}

\begin{theorem}[principio di Cavalieri]
  \label{th:cavalieri}
Sia $E \subset \RR^n\times \RR^m$ un insieme PJ-misurabile 
tale che per ogni $\vec x \in \RR^n$ la sezione 
$  E_{\vec x} = \ENCLOSE{\vec y \in \RR^m \colon (\vec x, \vec y) \in E} $ 
sia PJ-misurabile in $\RR^m$.
Allora
\[
  \abs{E} = \int_{\RR^n} \abs{E_{\vec x}}\, d\vec x.
\]

In particolare se $E,F\subset \RR^n\times \RR^m$ sono PJ-misurabili 
e per ogni $x\in \RR^n$ si ha $\abs{E_{\vec x}} = \abs{F_{\vec x}}$,
allora $\abs{E} = \abs{F}$.
\end{theorem}
\begin{proof}
Si applica il teorema di Fubini-Tonelli alla funzione caratteristica 
$1_E\colon \RR^n\times \RR^m \to \RR$:
\begin{align*}
  \abs{E} &= \int_{\RR^n\times \RR^m} 1_E
  = \int_{\RR^n} \Enclose{\int_{\RR^m} 1_E(\vec x, \vec y)\, d\vec y}\, d\vec x\\
  &= \int_{\RR^n} \Enclose{\int_{\RR^m} 1_{E_{\vec x}}(\vec y)\, d\vec y}\, d\vec x
  = \int_{\RR^n} \abs{E_{\vec x}}\, d \vec x.
\end{align*}
\end{proof}

\begin{lemma}[disuguaglianza di Lagrange]
  \label{lm:disuguaglianza_lagrange}
Sia $\Omega$ un aperto di $\RR^n$ e sia $f\in C^1(\Omega,\RR)$.
Se $\vec x,\vec y \in E$ e $[\vec x,\vec y]$ (il segmento che congiunge 
$\vec x$ e $\vec y$) ﺣ۷ contenuto in $\Omega$, allora si ha
\[
  \abs{f(\vec x) - f(\vec y)} \le \sup_{\vec z \in [\vec x,\vec y]} \Abs{Df(\vec z)}{\vec x - \vec y}.
\]
\end{lemma}
\begin{proof}
Sia $\phi(t) = f(\vec y + t(\vec x - \vec y))$.
Allora $\phi$ ﺣ۷ di classe $C^1$ e si ha
\begin{align*}
\abs{f(\vec x) - f(\vec y)}
&= \abs{\phi(1) - \phi(0)}
= \abs{\int_0^1 \phi'(t)\, dt}\\
& \le \int_0^1 \abs{Df(\vec y + t(\vec x - \vec y))(\vec x - \vec y)}\, dt \\
& \le \sup_{\vec z \in [\vec x,\vec y]} \Abs{Df(\vec z)}{\vec x - \vec y}.
\end{align*}
\end{proof}

\begin{lemma}[condizione di lipschitz per le funzioni $C^1$]
  \label{lm:lipschitz_C1}
  Sia $\Omega$ un aperto di $\RR^n$ e sia $f\in C^1(\Omega,\RR^m)$.
  Per ogni compatto $K\subset \Omega$ 
  la funzione $f$ ﺣ۷ lipschitizata su $K$.
  Ovvero esiste $L$ (dipendente da $K$)
  tale che
  per ogni $\vec x,\vec y \in K$ si ha
  \[
    \abs{f(\vec x) - f(\vec y)} \le L \abs{\vec x - \vec y}.
  \]
\end{lemma}
%
\begin{proof}
Visto che $K$ ﺣ۷ compatto esiste $\delta>0$ 
tale che per ogni $x\in K$ si ha $B_{2\delta}(x)\subset \Omega$
(ﺣ۷ sufficiente osservare che 
la funzione distanza da $\partial \Omega$ ﺣ۷ continua e positiva su $K$
dunque ha un minimo positivo per il teorema di Weiestrass).
Sia $K_\delta = \ENCLOSE{\vec z\in \RR^n\colon \dist(\vec z, K) \le \delta}$.
Possiamo definire $M = \max\ENCLOSE{\abs{f(\vec x)}\colon x \in K_{\delta}}$
e $\lambda = \max\ENCLOSE{\norm{Df(\vec x)}\colon \vec x \in K_{\delta}}$,
in quanto $K_\delta$ ﺣ۷ compatto e sia $\abs{f}$ che $\abs{Df}$ sono continue.

Ora se $\vec x,\vec y \in K$ e $\abs{\vec x - \vec y} < \delta$
la palla centrata in $\vec x$ di raggio $\delta$ contiene anche $\vec y$ e dunque
il segmento $[\vec x, \vec y]$ ﺣ۷ contenuto in $K_\delta\subset \Omega$.
Grazie al lemma~\ref{lm:disuguaglianza_lagrange} si ottiene dunque 
\[
  \abs{f(\vec y) - f(\vec x)} \le \lambda \abs{\vec y - \vec x}.
\]

Se invece $\vec x,\vec y \in K$ e $\abs{\vec x - \vec y} \ge \delta$ allora,
banalmente,
\[
\abs{f(\vec y)-f(\vec x)}
\le 2 M 
\le \frac{2M}{\delta}\abs{\vec y - \vec x}.
\]

Prendendo $L = \max\ENCLOSE{\lambda, \frac{2M}{\delta}}$ 
si ottiene dunque che $f$ ﺣ۷ $L$-lipschitziana 
su $K$.
\end{proof}

\begin{lemma}[misura delle immagini lipschitziane]
  \label{lm:misura_lipschitz}
Sia $E\subset \RR^n$ ed $\phi\colon E \to \RR^n$ 
una funzione $L$-lipschitziana.
Allora 
\[
  m^*(f(E)) \le (\sqrt n L)^n m^*(E).
\]
(dove $m^*$ ﺣ۷ la misura esterna di Peano-Jordan).
\end{lemma}
\begin{proof}
Per $A\subset \RR^n$ definiamo il diametro di $A$ come
\[
 \diam A \defeq \sup\{\abs{\vec x - \vec y}\colon \vec x, \vec y \in A\}.
\]
E' facile verificare che se $\phi$ ﺣ۷ $L$-lipschitziana allora
$\diam \phi(A) \le L \diam A$ per ogni $A\subset \RR^n$.
Consideriamo un cubetto diadico $Q^N_{\vec k}$ che tocca $E$.
Il diametro di $Q_N_{\vec k}$ ﺣ۷ $\sqrt n \cdot 2^{-N}$ e dunque 
il diametro di $\phi(Q^N_{\vec k}\cap E)$ ﺣ۷ al piﺣﺗ $\sqrt n L \cdot 2^{-N}$.
Significa che $\phi(Q^N_{\vec k}\cap E)$ ﺣ۷ contenuto 
in un cubo di lato $\sqrt n L \cdot 2^{-N}$ 
che ha misura $(\sqrt n L)^n \cdot 2^{-nN}$.
E' immediato verificare che $m^N(A\cup B)\le m^N(A) + m^N(B)$ 
e quindi $m^*(A\cup B)\le m^*(A) + m^*(B)$.
Dunque si ottiene:
\[
  m^*(\phi(E)) \le \sum_{Q^N_{\vec k}\cap E \neq \emptyset} m^*(\phi(Q^N_{\vec k}\cap E))
  \le \sum_{Q^N_{\vec k}\cap E \neq \emptyset} (\sqrt n L)^n \cdot 2^{-nN}
  = (\sqrt n L)^n m^N(E).
\]
Passando al limite per $N\to +\infty$ si ottiene il risultato voluto.
\end{proof}

\begin{lemma}[misurabilitﺣ  delle immagini $C^1$]
\label{lm:misurabile_C1}
Sia $\Omega$ un aperto di $\RR^n$ e sia $\phi\in C^1(\Omega,\RR^n)$
tale che $\det \phi(\vec x) \neq 0$ per ogni $\vec x \in \Omega$.
Allora se $E\Subset \Omega$ 
(significa che $\bar E$ ﺣ۷ compatto e contenuto in $\Omega$)
si ha $\partial (\phi(E)) \subset \phi(\partial E)$ e dunque 
se $E$ ﺣ۷ PJ-misurabile anche $\phi(E)$ ﺣ۷ PJ-misurabile.
\end{lemma}
\begin{proof}
Per il teorema di invertibilitﺣ  locale 
se $\vec x \in \Omega$ essendo $\det D\phi(\vec x) \neq 0$ esiste un intorno $U$ di $\vec x$
tale che $\phi$ ﺣ۷ un diffeomorfismo da $U$ a $\phi(U)$.
In particolare $\phi(U)$ ﺣ۷ un intorno di $\phi(\vec x)$ 
e dunque $\phi$ manda punti interni ad $E$ 
in punti interni a $\phi(E)$.
D'altra parte $\partial(\phi(E))\subset \overline{\phi(E)} \subset \phi(\bar E)$
in quanto $\bar E$ ﺣ۷ compatto e $\phi$, essendo continua, manda compatti in compatti.
Significa che i punti di $\partial(\phi(E))$ 
possono solo provenire da $\partial E$ 
e dunque $\partial(\phi(E)) \subset \phi(\partial E)$.

Se $E$ ﺣ۷ PJ-misurabile allora $\partial E$ ha misura nulla.
Per il lemma~\ref{lm:lipschitz_C1} la funzione $\phi$ ﺣ۷ lipschitziana su $\bar E$
e per il lemma~\ref{lm:misura_lipschitz} $\phi(\partial E)$ ha misura nulla.
Significa che $\phi(E)$ ﺣ۷ PJ-misurabile.
\end{proof}

Una \emph{matrice di permutazione} ﺣ۷ una matrice che rappresenta 
una applicazione lineare $L\colon \RR^n\to \RR^n$ che permuta 
gli elementi della base canonica: 
$L (x_1,\dots,x_n) = (x_{\sigma(1)},\dots,x_{sigma(n)})$.
Una \emph{mossa elementare} su $\RR^n$ 
ﺣ۷ una applicazione lineare $L$ che aggiunge ad una componente del vettore 
$\vec x$ un multiplo fissato di un'altra coordinata:
$L(x_1,\dots, x_n) = (x_1, \dots, x_j + \lambda x_k, \dots x_n)$
con $k\neq j$.
Chiamiamo \emph{matrice elementare} una matrice che si ottiene 
come prodotto di matrici di permutazione e mosse elementari.
Visto che le matrici di permutazione hanno determinante $\pm 1$
e le mosse elementari hanno sempre determinante $1$, 
tutte le matrici elementari hanno determinante $\pm 1$.

\begin{definition}[matrice elementare]
Dati $k,j\in \{1,\ldots,n\}$ con $k\neq j$ e $\alpha \in \RR$
consideriamo la matrice $M=M(k,j,\alpha)$ 
definita sulla base canonica $e_1,\dots, e_n$ da 
\[
   M e_i = \begin{cases}
      e_i & \text{se $i\neq k$}\\
      e_i + \alpha e_j & \text{se $i=k$}.
   \end{cases}
\]
Ovvero $M$ ﺣ۷ una matrice che si ottiene mettendo il valore $\alpha$
nella matrice identitﺣ  in una posizione dove c'era uno zero:
\[
   M = \begin{pmatrix}
    1 & 0 & \dots & 0 & \dots & 0\\
    0 & 1 & \dots & \alpha & \dots & 0\\
    \vdots & \vdots & \ddots & \vdots & \ddots & \vdots\\
    0 & 0 & \dots & 1 & \dots & 0\\
    \vdots & \vdots & \ddots & \vdots & \ddots & \vdots\\
    0 & 0 & \dots & 0 & \dots & 1
   \end{pmatrix}.
\]
In altre parole la trasformazione lineare associata ad $M$ 
agisce sullo spazio generato dai due vettori $e_j$ ed $e_k$
come la matrice $2\times 2$
\[
  \begin{pmatrix}
    1 & \alpha\\
    0 & 1
  \end{pmatrix}
\]
e come l'identitﺣ  su tutti gli altri vettori della base canonica.

Quando la matrice $M$ viene moltiplicata a sinistra 
di una matrice $A$ di dimensione $m\times n$
si ottiene una matrice $M A$ che ﺣ۷ la matrice ottenuta da $A$
aggiungendo $\alpha$ volte la $j$-esima riga di $A$
alla $k$-esima riga di $A$.

Viceversa quando la matrice $M$ viene moltiplicata a destra ad una matrice $A$ di dimensione $m\times n$
si ottiene una matrice $A M$ che si ottiene da $A$ aggiungendo $\alpha$ volte la $k$-esima colonna di $A$
alla $j$-esima colonna di $A$.

Una \emph{matrice di permutazione} ﺣ۷ una matrice $M$ che permuta gli elementi 
della base canonica. 
Piﺣﺗ precisamente data una bigezione $\sigma: \ENCLOSE{1,\ldots,n} \to \ENCLOSE{1,\ldots,n}$, 
si ha $M(e_k) = e_{\sigma(k)}$ per ogni $k=1,\ldots,n$.

Una \emph{matrice elementare} ﺣ۷ una matrice che si ottiene come prodotto di matrici di permutazione e mosse elementari.
Le matrici elementari sono banalmente invertibili 
ed hanno tutte determinante $\pm 1$.
\end{definition}

\begin{theorem}[riduzione a scala di Gauss]
\label{th:riduzione_gauss}
Sia $M$ una matrice $m\times n$ qualunque. 

Allora esistono $A$ matrice elementare $m\times m$,
e $B$ matrice elementare $n\times n$ tali che
\[
  A M B^t = \Delta
\]
ﺣ۷ una matrice diagonale.
\end{theorem}

\begin{theorem}[significato geometrico del determinante]
  \label{th:geometria_determinante}
Sia $L\colon \RR^n\to \RR^n$ una applicazione lineare affine
\[
  L(\vec x) = M\vec x + \vec q.
\]
Allora se $E\subset \RR^n$ ﺣ۷ PJ-misurabile anche $L(E)$ ﺣ۷ PJ-misurabile 
e si ha 
\begin{equation}\label{eq:geometria_determinante}
  \abs{L(E)} = \abs{\det M} \cdot \abs{E}.
\end{equation}
\end{theorem}
\begin{proof}

\emph{Osservazione 1.}
Fissato $L$ se dimostriamo 
\eqref{eq:geometria_determinante}
su ogni cubetto $Q^N_{\vec k}$, 
allora approssimando opportunamente 
$E$ dall'interno e dall'esterno con cubetti di taglia $2^{-N}$, 
e usando l'additivitﺣ  della misura,
si ottiene
\[
  \abs{\det M} \cdot m_N(L(E)) \le m^N(L(E)) \le \abs{\det M}\cdot m^N(E).
\]
Passando al limite per $N\to +\infty$ si ottiene il risultato 
voluto con tale $L$.

\emph{Osservazione 2.}
Se per ogni $E$ l'equazione 
\eqref{eq:geometria_determinante} ﺣ۷ verificata per $L_1$ e per $L_2$ 
allora ﺣ۷ verificata anche per 
$L_1\circ L_2$. 
Infatti
\[
  \abs{L_1(L_2(E))} = \abs{\det M_1} \cdot \abs{L_2(E)} = \abs{\det M_1} \cdot \abs{\det M_2} \cdot \abs{E}
  = \abs{\det(M_1M_2)}\cdot\abs{E}.
\]

\emph{Passo 0: supponiamo $M$ sia la matrice identitﺣ .}
In tal caso $L$ ﺣ۷ una traslazione e dunque $L(E) = E + \vec q$.
Se $E$ ﺣ۷ un cubetto $Q^N_{\vec k}$ allora $L(E)$ 
ﺣ۷ un cubetto di lato $2^{-N}$ allineato agli assi coordinati.
Dunque $\abs{L(E)} = 2^{-nN} = \abs{E}$ per l'esercizio~\ref{ex:misura_rettangolo}.

\emph{Passo 1: supponiamo che $M$ sia una matrice di permutazione
(cioﺣ۷ $M(e_k) = e_{\sigma(k)}$ per una qualche permutazione $\sigma$ di
$\{1,\ldots,n\}$)}.
Per l'osservazione possiamo supporre che $E$ sia un cubetto $Q^N_{\vec k}$.
In tal caso il risultato ﺣ۷ immediato in quanto $L(Q^N_{\vec k}) = Q^N_{M(\vec k)}+\vec q$,
dunque $\abs{L(E)} = 2^{-nN} = \abs{E}$ (per l'esercizio~\ref{ex:misura_rettangolo})
Visto che $\abs{\det M} = 1$ questo ﺣ۷ il risultato voluto. 

\emph{Passo 2: supponiamo che $M$ sia una matrice diagonale
(cioﺣ۷ $M(e_k) = \lambda_k e_k$ per ogni $k=1,\ldots,n$) e $\vec q=0$}.
Supponendo che $E$ sia un rettangolo $[a_1,b_1]\times \dots \times [a_n,b_n]$,
si osserva che 
\[
  L(E) = \prod_{k=1}^n [\lambda_k a_k, \lambda_k b_k]
\]
(se $\lambda_k < 0$ gli estremi del rettangolo sono invertiti).
Dunque per l'esercizio~\ref{ex:misura_rettangolo} si ha
\[
  \abs{L(E)} = \prod_{k=1}^n \abs{\lambda_k}\cdot \abs{b_k-a_k}
  = \abs{\prod_{k=1}^n \lambda_k } \cdot \prod_{k=1}^n \abs{b_k - a_k} 
  = \abs{\det M} \cdot \abs{E}.
\]

\emph{Passo 3: supponiamo che $L$ sia della forma 
$L(x_1,\dots,x_n)=(x_1,\dots,x_{n-2}, x_{n-1} + \lambda x_n, x_n)$}
In questo caso $L(Q^N_{\vec k})$ ﺣ۷ un parallelepipedo con stessa base 
e stessa altezza di $Q^N_{\vec k}$ e dunque, per il teorema~\ref{th:cavalieri}
di Cavalieri si ottiene $\abs{L(Q^N_{\vec k})} = \abs{Q^N_{\vec k}}$.
Chiaramente si ha $\det L = 1$ e quindi questo ﺣ۷ quanto volevamo dimostrare.
Piﺣﺗ precisamente si puﺣﺎ osservare che posto, per $t\in \RR$,
\[
  (L(Q^N_{\vec k}))_t 
  = \ENCLOSE{\vec y \in \RR^{n-1} \colon (\vec y,t) \in L(Q^N_{\vec k})},
\]
e posto $\vec k' = (k_1,\ldots,k_{n-1})\in \ZZ^{n-1}$ si ha
$(\vec y,t) \in L(Q^N_{\vec k})$ se e solo se
$t\in [2^{-N} k_n, 2^{-N}(k_n+1)]$ e
$\vec y \in Q^N_{\vec k'} + \lambda t e_{n-1}$.
Dunque
\[
  (L(Q^N_{\vec k}))_t 
  = Q^N_{\vec k'} + t e_{n-1}
\]
e $\abs{(L(Q^N_{\vec k}))_t} = 2^{-(n-1)N}$.
Dunque, per il principio di Cavalieri, si ottiene
\[
  \abs{L(Q^N_{\vec k})}
  = \int_{2^{-N} k_n}^{2^{-N}(k_n+1)} \abs{(L(Q^N_{\vec k}))_t}\, dt
  = 2^{-N} \cdot 2^{-(n-1)N} 
  = 2^{-nN} = \abs{Q^N_{\vec k}}.
\]

\emph{Caso generale.}
Grazie al teorema~\ref{th:riduzione_gauss} di riduzione a scala di Gauss,
possiamo trovare
$A,B$ matrici elementari tali che
$AMB^t$ 
sia una matrice diagonale $\Delta$. 
Ovvero
\[
   M = A^{-1} \Delta (B^t)^{-1}.
\]
Visto che abbiamo giﺣ  dimostrato che l'equazione~\eqref{eq:geometria_determinante} ﺣ۷ verificata per $A^{-1}$, per $\Delta$ e per $(B^t)^{-1}$,
dall'osservazione 2 si ottiene che ﺣ۷ verificata anche per $M$ e dunque per $L$.
\end{proof}

\begin{theorem}[formula dell'area]
  \label{th:formula-area}
Sia $\Omega\subset \RR^n$ un aperto, 
$\phi\colon \bar \Omega \to \RR^n$ una funzione di 
classe $C^1$  
tale che $\phi$ ﺣ۷ iniettiva su $\Omega$ 
e $\det D \phi(\vec x)\neq 0$ per ogni $\vec x\in \Omega$.

Allora per ogni $E\subset \Omega$ limitato 
e PJ-misurabile si ha
che $\phi(E)$ ﺣ۷ PJ-misurabile e 
\[
  \abs{\phi(E)} 
    = \int_{E} \abs{\det D\phi}.
\]
\end{theorem}

\begin{proof}
\emph{Passo 1: misurabilitﺣ .} 
Il lemma~\ref{lm:misurabile_C1} ci dice che $\phi(E)$ ﺣ۷ PJ-misurabile.

\begin{figure}
\centering \includegraphics[height=2cm]{cubo-interno.png}
\end{figure}
\emph{Passo 2 (lemma del cubo interno/esterno): 
se $E = [0,1]^n \subset \Omega$  
ed esiste $\eps\in(0,\frac 1 4)$ tale che 
\[
  \abs{\phi(\vec x)-\vec x} \le \eps, \qquad \forall \vec x \in E,
\]
allora }
\[
  (1-2\eps)^n
  \le \abs{\phi(E)}
  \le (1+2\eps)^n.
\]
Chiaramente 
$
    \phi(E) \subset [-\eps,1+\eps]^n
$
perchﺣ۸ ogni punto di $E$ viene spostato al massimo di $\eps$. 
Quindi si ottiene, per monotonia della misura,
$\abs{\phi(E)} \le (1+2\eps)^n$.
Per ottenere l'altra disuguaglianza  
$\abs{\phi(E)} \ge (1-2\eps)^n$
basterﺣ  dimostrare che vale anche 
l'inclusione opposta
$\phi(E) \supset (\eps,1-\eps)^n$.

Supponiamo per assurdo che esista $\vec y \in (\eps,1-\eps)^n \setminus \phi(E)$.
Consideriamo il punto $\vec p=(\frac 12, \dots, \frac 12)$ centro 
del cubo $E$ e sia $\vec q=\phi(\vec p)$ la sua immagine. 
Visto che $\abs{\vec q-\vec p} = \abs{\phi(\vec p)-\vec p} \le \eps$ 
e $\eps \le \frac 1 4$ possiamo garantire che
$\vec q \in (\eps,1-\eps)^n$.
Dunque l'intero segmento $[\vec q,\vec y]$ ﺣ۷ contenuto 
in $(\eps,1-\eps)^n$ essendo quest'ultimo un convesso.
Visto che $\vec q\in \phi(E)$ mentre $\vec y \notin \phi(E)$,
deve esistere un punto $\vec z\in [\vec q, \vec y]\cap \partial \phi(E)$
(esercizio~\ref{ex:punto-di-frontiera}).
Ma abbiamo giﺣ  mostrato (Passo 1) che $\partial \phi(E) \subset \phi(\partial E)$
dunque $\vec z=\phi(\vec x)$ con $\vec x \in \partial E$. 
Ma allora $\abs{\vec z-\vec x} = \abs{\phi(\vec x)- \vec x} \le \eps$ 
e dunque
$\vec z\not \in (\eps,1-\eps)^n$, 
contraddicendo $\vec z \in [\vec q, \vec y] \subset (\eps,1-\eps)^n$.

\emph{Conclusione.}
Sia $\eps>0$ fissato.
Sia $A$ un aperto tale che  $E \Subset A \Subset \Omega$.
Dato $N\in \NN$ definiamo $E^N$ l'unione dei cubetti 
diadici $Q^N_{\vec k}$ che intersecano $E$ 
e $E_N$ l'unione dei cubetti diadici $Q^N_{\vec k}$ che sono contenuti in $E$.
Affermiamo che ﺣ۷ possibile scegliere $N$ sufficientemente grande 
affinchﺣ۸:
\begin{gather}
  \label{eq:shme1}
  E^N \subset A,\\
  \label{eq:shme2}
  \abs{\vec x - \vec x'}\le \sqrt{n}2^{-N} \implies \Abs{(D\phi(\vec x))^{-1} D\phi(\vec x') - I} \le \frac{\eps}{\sqrt{n}}.
\end{gather}
La condizione~\eqref{eq:shme2} si ottiene 
grazie all'uniforme continuitﺣ  di $D\phi$ su $\bar A$.

Fissato un cubetto $Q^N_{\vec k}\subset E$ 
consideriamo la traformazione $\psi$ che prende 
il cubo $[0,1]^n$ lo manda tramite un riscalamento in $Q^N_{\vec k}$,
applica $\phi$ e poi riporta l'immagine in $[0,1]^n$
tramite l'inversa di una mappa affine che approssima $\phi$
e un riscalamento inverso.
Precisamente $\psi\colon[0,1]^n \to \RR^n$ ﺣ۷ definita da

\begin{proposition}[interno in interno per mappe localmente iniettive]
Se $E\subset \RR^n$ e $\phi\colon E\to \RR^n$ ﺣ۷ continua e localmente iniettiva sull'interno di $E$, allora $\phi(\operatorname{int} E)\subset \operatorname{int} \phi(E)$.
In particolare, se $\phi\in C^1(E)$ e $\det D\phi\ne 0$ su $\operatorname{int}E$, allora $\phi$ manda punti interni in punti interni.
\end{proposition}
\begin{proof}
Sia $x\in \operatorname{int}E$. Per localmente iniettiva, esiste $\eps>0$ tale che la palla chiusa $\overline B_\eps(x)\subset \operatorname{int}E$ e $\phi$ sia iniettiva su $\overline B_\eps(x)$.
La restrizione $\phi\colon \overline B_\eps(x)\to \RR^n$ ﺣ۷ una bigezione continua da un compatto in uno spazio di Hausdorff, dunque ﺣ۷ un omeomorfismo sulla sua immagine $L\defeq \phi(\overline B_\eps(x))$, che ﺣ۷ compatta.
Perciﺣﺎ $\partial L = \phi(\partial B_\eps(x))$. Poichﺣ۸ $\phi(x)\notin \phi(\partial B_\eps(x))$ e $\partial B_\eps(x)$ ﺣ۷ compatta, la distanza
\[
  \delta \defeq \dist\bigl(\phi(x),\partial L\bigr) > 0.
\]
Ne segue che $B_\delta(\phi(x))\subset L\subset \phi(E)$, cioﺣ۷ $\phi(x)\in \operatorname{int}\phi(E)$. Il caso $C^1$ con $\det D\phi\ne 0$ discende dal teorema dell'inversa, che garantisce l'iniettivitﺣ  locale.
\end{proof}

\begin{theorem}[cambio di variabile]
  \label{th:cambio-variabili}
Siano $\Omega\subset \RR^n$ e $\phi\colon \bar \Omega \to \RR^n$ 
come nelle ipotesi del teorema~\ref{th:formula-area}
e sia $f\colon \phi(\Omega)\to\RR$ una funzione
Riemann-integrabile, limitata.
Allora
\[

  =
  \int_\Omega f(\phi(\vec x))\cdot \abs{\det D \phi(\vec x)}\, d \vec x.
\]

La formula si memorizza mediante la sostituzione formale:
\[
\begin{cases}
  \vec y = \phi(\vec x),\\
  d \vec y = \abs{\det D\phi(\vec x)}\, d\vec x.
\end{cases}
\]
\end{theorem}
