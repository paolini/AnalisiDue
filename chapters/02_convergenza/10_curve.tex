\chapter{curve e lunghezza}

Sia $X$ uno spazio metrico. Una funzione $u\colon [a,b] \to X$
viene chiamata \emph{curva}.

\begin{definition}[lunghezza di una curva]
Diremo che $P$ ﺣ۷ una \emph{suddivisione} di $[a,b]$ se
$P$ ﺣ۷ un sottoinsieme finito di $[a,b]$ con $a,b\in P$.
Dunque $P = \{ t_0, t_1, \dots, t_N\}$ con $a= t_0 < t_1 < \dots < t_N = b$.
Ad ogni suddivisione $P$ di $[a,b]$ possiamo associare la
lunghezza della \emph{poligonale} $u(t_0), u(t_1), \dots, u(t_N)$ cioﺣ۷
\[
  \ell_P(u) \defeq \sum_{k=1}^N d(u(t_{k-1}), u(t_k)).
\]
Definiamo la
\emph{lunghezza} della curva $u$ come
\[
    \ell(u) \defeq \sup \ENCLOSE{ \ell_P(u) \colon P \text{ suddivisione di } [a,b] }.
\]
Spesso sarﺣ  utile considerare la funzione $u$ ristretta ad un sottointervallo $[\alpha,\beta]\subset [a,b]$.
Useremo la notazione
\[
  \ell_\alpha^\beta(u) \defeq \ell(u \llcorner [\alpha,\beta])
\] 
cosicchﺣ۸ $\ell(u) = \ell_a^b(u)$.

In generale $\ell(u)\in [0, +\infty]$.
Se $\ell(u) < +\infty$, diremo che la curva ﺣ۷ \emph{rettificabile}
ovvero che ha \emph{lunghezza finita}.
\end{definition}

\begin{theorem}[proprietﺣ  della lunghezza]
Sia $\vec u \colon [a,b] \to X$ una curva in uno spazio metrico $(X,d)$.
Allora valgono le seguenti proprietﺣ .
\begin{enumerate}
\item (disuguaglianza triangolare) $\ell(u) \ge d(u(a), u(b))$;
\item (additivitﺣ ) per ogni $c\in [a,b]$ si ha 
  $\ell_a^b(u) = \ell_a^c(u) + \ell_c^b(u)$.
\item (semicontinuitﺣ )
se $u_n\colon [a,b] \to X$ converge puntualmente a $u$
(cioﺣ۷ $u_n(t) \to u(t)$ per ogni $t\in[a,b]$)
allora 
\[
  \ell(u) \le \liminf_{n\to \infty} \ell(u_n).
\]
\end{enumerate}
\end{theorem}
\begin{proof}
La prima proprietﺣ  (disuguaglianza triangolare) ﺣ۷ ovvia 
considerando la suddivisione minima $P = \{ a,b\}$ 
per cui $\ell(u) \ge \ell_P(u) = d(u(a), u(b))$.

Dimostriamo la seconda proprietﺣ  (additivitﺣ ).
Data una suddivisione $P$ di $[a,c]$ e una suddivisione $Q$ di $[c,b]$,
la loro unione $P\cup Q$ ﺣ۷ una suddivisione di $[a,b]$. Dunque 
\[
    \ell_a^b(u) \ge \ell_{P\cup Q}(u) = \ell_{P}(u) + \ell_{Q}(u)
\] 
da cui, per l'abitrarietﺣ  di $P$ e $Q$ si ottiene 
la disuguaglianza $\ell_a^b(u) \ge \ell_a^c(u) + \ell_c^b(u)$.
Viceversa se $P$ ﺣ۷ una suddivisione di $[a,b]$ possiamo 
considerare le due suddivisioni $P_1 = P \cap [a,c] \cup \ENCLOSE c$ e
$P_2 = P \cap [c,b] \cup \ENCLOSE c$ 
e ottenere 
\[
\ell_P(u) \le \ell_{P\cup\ENCLOSE c} 
= \ell_{P_1}(u) + \ell_{P_2}(u) 
\le \ell_a^c(u) + \ell_c^b(u).
\]
Dall'arbitrarietﺣ  di $P$ si ottiene la disuguaglianza opposta
$\ell_a^b(u) \le \ell_a^c(u) + \ell_c^b(u)$.

Dimostriamo la semicontinuitﺣ .
Fissata una qualunque suddivisione $P={t_0,\dots, t_N}$
per ogni $k=0,\dots, N$ si ha $u_n(t_k)\to u(t_k)$ per $n\to +\infty$ e quindi:
\[
  \ell_P(u) =
  \lim_n \ell_P(u_n)
  = \liminf_n \ell_P(u_n)
  \le \liminf_n \ell(u_n).
\]
Dall'arbitrarietﺣ  di $P$ si ottiene la tesi.
\end{proof}

\begin{theorem}[formula della lunghezza]
    Sia $\vec u \colon [a,b] \to \RR^n$ una curva di classe $C^1$.
    Allora la curva ﺣ۷ rettificabile e la sua lunghezza ﺣ۷ data da
    \[
      \ell(\vec u) = \int_a^b \abs{\vec u'(t)} \, dt.
    \]
\end{theorem}
\begin{proof}
Sia $a = t_0 < t_1 < \dots < t_N = b$ una suddivisione $P$ di $[a,b]$.
Osserviamo che 
\[
  \vec u(t_{k+1}) - \vec u(t_k) = \int_{t_k}^{t_{k+1}} \vec u'(t) \, dt
\]
da cui 
\[
 \abs{\vec u(t_{k+1})-\vec u(t_k)} \le \int_{t_k}^{t_{k+1}} \abs{\vec u'(t)} \, dt
\]
e quindi 
\[
  \ell_P(\vec u) \le \int_a^b \abs{\vec u'(t)} \, dt.
\]
Dall'arbitrarietﺣ  di $P$ si ottiene la prima disuguaglianza 
$\ell(\vec u) \le \int_a^b \abs{\vec u'}$.

D'altra parte $\vec u'$ ﺣ۷ continua in $[a,b]$ dunque ﺣ۷ uniformemente continua.
Dato $\eps>0$ esiste $\delta>0$ tale che
\[
  \abs{t-s} < \delta \implies \abs{\vec u'(t) - \vec u'(s)} < \eps.
\]
Visto che $\abs{\vec u'}$ ﺣ۷ integrabile (in quanto continua) esiste una suddivisione
$a = t_0 < t_1 < \dots < t_N = b$ con $t_{k+1}-t_k < \delta$ per cui
\[
  \abs{\int_a^b \abs{\vec u'(t)} \, dt - \sum_{k=0}^{N-1} \inf_{t\in [t_k, t_{k+1}]} \abs{\vec u'(t_k)} (t_{k+1}-t_k)} 
  < \eps.
\]
Allora 
\begin{align*}
  \int_a^b \abs{\vec u'(t)} \, dt
    &\le \sum_{k=0}^{N-1} \abs{\vec u'(t_k)} (t_{k+1}-t_k) + \eps
    = \sum_{k=0}^{N-1} \abs{ \int_{t_k}^{t_{k+1}} \vec u'(t_k) \, dt} + \eps \\
    &\le \sum_{k=0}^{N-1} \Enclose{\int_{t_k}^{t_{k+1}} \abs{\vec u'(t_k) - \vec u'(t)} \, dt + \abs{\int_{t_k}^{t_{k+1}} \vec u'(t)\, dt}} + \eps \\
    &\le \sum_{k=0}^{N-1} \Enclose{ \eps (t_{k+1}-t_k) + \abs{\vec u(t_{k+1}) - \vec u(t_k)}} + \eps \\
    &=  \eps + \sum_{k=0}^{N-1}  \abs{\vec u(t_{k+1}) - \vec u(t_k)} + \eps
    \le \ell(\vec u) + 2\eps.
\end{align*}
Dall'arbitrarietﺣ  di $\eps>0$ si ottiene la disuguaglianza opposta e la tesi.
\end{proof}

\section{riparametrizzazioni}

Se $u\colon [a,b] \to X$ ﺣ۷ una curva e $\phi\colon [A,B] \to [a,b]$ ﺣ۷ una funzione, 
possiamo considerare la curva $v\colon [A,B] \to X$ definita da
\[
  v(s) = u(\phi(s)), \qquad s\in [A,B].
\]
Se $u$ ﺣ۷ continua e $\phi$ ﺣ۷ continua e crescente, la curva $v$ rappresenta 
lo stesso tragitto di $u$ percorso perﺣﺎ con una diversa velocitﺣ . 
I punti attraversati sono gli stessi, nello stesso ordine, ma l'istante in cui 
vengono raggiunti ﺣ۷ diverso.
Se $\phi:[A,B] \to [a,b]$ ﺣ۷ strettamente crescente e continua
allora $\phi$ ﺣ۷ invertibile e si possono mettere in corrispondenza biunivoca 
i parametri delle due curve $u(t)$ e $v(s)$ ponendo $t=\phi(s)$ per cui:
\[
  v(s) = u(\phi(s)) \qquad \text{e} \qquad u(t) = v(\phi^{-1}(t)).
\]
Risulta perﺣﺎ utile considerare anche riparametrizzazioni che non sono 
strettamente crescenti o che non sono continue.
Ad esempio se $u$ ha dei tratti costanti, ha senso considerare una riparametrizzazione 
che salta i tratti costanti, senza fermarsi su di essi. 
Una tale riparametrizzazione presenterﺣ  dei salti, 
ma il tragitto percorso ﺣ۷ in effetti lo stesso. 
Viceversa puﺣﺎ essere utile considerare una riparametrizzazione che si ferma 
per un certo tempo su un punto: tale riparametrizzazione avrﺣ  un tratto costante 
e quindi non sarﺣ  strettamente crescente e non sarﺣ  invertibile.

Se una riparametrizzazione non ﺣ۷ continua puﺣﺎ saltare dei tratti 
e quindi puﺣﺎ dare origine ad una curva di lunghezza inferiore.
Se la riparametrizzazione non ﺣ۷ monotona puﺣﺎ ripercorrere 
piﺣﺗ volte gli stessi tratti e quindi puﺣﺎ dare origine ad una curva di lunghezza 
superiore. Questo viene formalizzato nel seguente teorema. 

\begin{theorem}[riparametrizzazione]\label{th:riparametrizzazione}
Sia $u \colon [a,b] \to X$ una curva in uno spazio metrico $X$ 
e sia $\phi \colon [A,B] \to [a,b]$ una funzione.
Possiamo considerare la curva $v \colon [A,B] \to X$ definita da
$v=u \circ \phi$, cioﺣ۷
\[
  v(s) = u(\phi(s)), \qquad s\in [A,B].
\]

\begin{enumerate}
\item
Se $\phi$ ﺣ۷ monotona si ha
\[
  \ell(v) \le \ell(u).
\]
\item Se $\phi\colon [A,B] \to [a,b]$ ﺣ۷ continua e surgettiva si ha
\[
  \ell(v) \ge \ell(u).
\]
\item se $\phi\colon [A,B] \to [a,b]$ ﺣ۷ bigettiva e continua
si ha:
\[
  \ell(u) = \ell(v).
\]
\end{enumerate}
\end{theorem}
%
\begin{proof}
Dimostriamo il primo punto, dove si suppone che $\phi:[A,B] \to [a,b]$ 
sia monotona. Per fissare le idee supponiamo $\phi$ sia crescente.
Allora presa una qualunque suddivisione $Q$ di $[A,B]$ si ha che 
$P=\phi(Q)\cup \ENCLOSE{a,b}$ ﺣ۷ una suddivisione di $[a,b]$.
Se i punti di $Q$ sono $A = t_0 < t_1 < \dots < t_N = B$ i punti di 
$P$ sono $a \le \phi(t_0) \le \phi(t_1) \le \dots \le \phi(t_N) \le b$.
Osserviamo dunque che alcuni punti consecutivi di questo elenco 
potrebbero coincidere
(se $\phi$ non ﺣ۷ iniettiva) ma se si ha $\phi(t_k) = \phi(t_{k+1})$
si ha, ovviamente, $d(u(\phi(t_k)), u(\phi(t_{k+1}))) = 0$. 
Dunque risulta comunque 
\begin{align*}
 \ell_Q(v) &= \sum_{k=1}^N d(v(t_{k-1}), v(t_k))
  = \sum_{k=1}^N d(u(\phi(t_{k-1})), u(\phi(t_k)))\\
  &\le d(u(a), u(t_0)) + \sum_{k=1}^N d(u(\phi(t_{k-1})), u(\phi(t_k))) + d(u(t_N), u(b))\\
  &\le \ell_P(u) \le \ell(u).
\end{align*}
Dall'arbitrarietﺣ  di $Q$ si ottiene la disuguaglianza $\ell(v) \le \ell(u)$.

Vediamo il caso in cui $\phi$ ﺣ۷ continua e surgettiva. 
In questo caso prendiamo una qualunque suddivisione $P$ di $[a,b]$ 
formata dai punti $a = t_0 < t_1 < \dots < t_N = b$.
Vogliamo ora dimostrare che esiste una suddivisione $Q$ di $[A,B]$
formata da punti $A = s_0 < s_1 < \dots < s_N = B$ tali 
che $\phi(s_k) = t_k$ per ogni $k=0,\dots,N$. 
Si procede induttivamente. Prendiamo $s_0=A$ e osserviamo che $\phi(s_0) = t_0$
in quanto per ipotesi $\phi(A) = a$.
Una volta definiti $A = s_0 < s_1 < \dots < s_n < B$ tali che $\phi(s_k) = t_k$ per $k=0,\dots,n$,
andiamo a definire $s_{n+1}$ se $n < N$.
Visto che $\phi(s_n) = t_n < t_{n+1} \le b = \phi(B)$ 
per il teorema dei valori intermedi esiste $s_{n+1} \in (s_n, B]$ tale che
$\phi(s_{n+1}) = t_{n+1}$. 
Procedendo in questo modo si ottiene la suddivisione $Q$ di $[A,B]$ 
con le proprietﺣ  richieste e come ultimo punto $s_N$ si puﺣﺎ senz'altro scegliere $B$ 
in quanto $\phi(B) = b = t_N$.
Dunque si ha 
\[
  \ell_P(u) = \sum_{k=1}^N d(u(t_{k-1}), u(t_k))
  = \sum_{k=1}^N d(v(s_{k-1}), v(s_k)) = \ell_Q(v) \le \ell(v).
\]
Dall'arbitrarietﺣ  di $P$ si ottiene la disuguaglianza voluta $\ell(u) \le \ell(v)$.

Il terzo punto segue immediatamente dai primi due, in quanto una funzione iniettiva e continua 
definita tra due intervalli ﺣ۷ necessariamente monotona (per il teorema dei valori intermedi).
\end{proof}

\begin{definition}[lunghezza d'arco]
Sia $u \colon [a,b] \to X$ una curva rettificabile in uno spazio metrico $X$.
Sia $L = \ell(u)$ la lunghezza della curva.
La \emph{lunghezza d'arco} di $u$ ﺣ۷ la funzione $\sigma\colon [a,b] \to [0,L]$ 
che mi dﺣ  la lunghezza percorsa dalla curva $u$ 
fino al tempo $t$, formalmente ﺣ۷ definita da
\begin{equation}\label{def:lunghezza_arco}
  \sigma(t) \defeq \ell_a^t(u).
\end{equation}
\end{definition}

Chiaramente la funzione $\sigma$ ﺣ۷ crescente ma potrebbe non essere strettamente crescente 
(e quindi non iniettiva) se la curva $u$ ha degli intervalli su cui ﺣ۷ costante.

\begin{theorem}[continuitﺣ  della lunghezza d'arco]
Sia $u \colon [a,b] \to X$ una curva continua rettificabile, di lunghezza $L$, 
in uno spazio metrico $X$.
Allora la funzione lunghezza d'arco $\sigma\colon [a,b] \to [0,L]$,
$\sigma(t) = \ell_a^t(u)$,
ﺣ۷ continua.
\end{theorem}
\begin{proof}
Fissato $t_0\in [a,b)$
mostreremo che $\sigma(t)\to \sigma(t_0)$ per $t\to t_0^+$. Il limite sinistro $t\to t_0^-$ 
quando $t_0\in(a,b]$ si tratta 
in maniera del tutto analoga.

Osserviamo che, per additivitﺣ  della lunghezza, si ha 
$\sigma(t) - \sigma(t_0) = \ell_a^t(u) - \ell_a^{t_0}(u) = \ell_{t_0}^t(u)$.
Dobbiamo dunque dimostrare che $\ell_{t_0}^t(u) \to 0$ per $t\to t_0^+$.

Fissato $\eps>0$ dalla rettificabilitﺣ  di $u$ 
possiamo trovare dei punti $t_1, t_2, \dots, t_N \in (t_0,b]$
tali che $t_0 < t_1 < t_2 < \dots < t_N$ e
\[
\ell_{t_0}^b(u) 
\le \sum_{k=1}^N d(u(t_{k-1}), u(t_k)) + \eps
\]
da cui
\begin{align*}
\ell_{t_0}^b(u) 
   &\le d(u(t_0), u(t_1)) + \sum_{k=2}^N d(u(t_{k-1}), u(t_k)) + \eps \\
   &\le d(u(t_0), u(t_1)) + \ell_{t_1}^b(u) + \eps
\end{align*}
ovvero
\[
  \ell_{t_0}^{t_1}(u) = \ell_{t_0}^b(u) - \ell_{t_1}^b(u) \le d(u(t_0), u(t_1)) + \eps.
\]
Scelto ora un qualunque punto $t\in (t_0, t_1)$ si ha
\begin{align*}
  \ell_{t_0}^t(u) 
    &= \ell_{t_0}^{t_1}(u) - \ell_t^{t_1}(u) \\
    &\le d(u(t_0), u(t_1)) + \eps - \ell_t^{t_1}(u) \\
    &\le d(u(t_0), u(t)) + d(u(t), u(t_1)) + \eps - \ell_t^{t_1}(u) \\
    &\le d(u(t_0), u(t)) + \eps.
\end{align*}
Essendo $u$ continua si ha $d(u(t_0), u(t)) \to 0$ per $t \to t_0$ 
e dunque
\[
\limsup_{t\to t_0^+} \ell_{t_0}^t(u) \le \eps.
\]
Ma questo ﺣ۷ vero per ogni $\eps>0$, dunque si ottiene, come volevamo dimostrare,
\[
\lim_{t\to t_0^+} \ell_{t_0}^t(u) = 0.
\]
\end{proof}

Grazie al teorema precedente sappiamo che se $u\colon [a,b] \to X$ ﺣ۷ una curva continua e rettificabile, 
allora la sua lunghezza d'arco $\sigma\colon[a,b] \to [0,L]$ ﺣ۷ monotona e continua.
Dunque possiamo definire una \emph{inversa sinistra di $\sigma$} 
come la funzione $\tau\colon [0,L] \to [a,b]$ definita da
\[
  \tau(s) = \min \ENCLOSE{t\colon \sigma(t)=s}.
\]
La curva riparametrizzata $\tilde u\colon [0,L] \to X$ definita da:
\[
  \tilde u(s) = u(\tau(s))
\]
si chiama \emph{riparametrizzazione per lunghezza d'arco} di $u$.
La funzione $\tau$ ﺣ۷ strettamente crescente ma puﺣﺎ avere delle discontinuitﺣ  
di tipo salto. Se $s_0$ ﺣ۷ un punto in cui $\tau$ ha un salto denotiamo con 
$\tau(s_0^-)$ e $\tau(s_0^+)$ i limiti sinistro e destro di $\tau$ in $s_0$.
Posto $I = [\tau(s_0^-), \tau(s_0^+)]$ 
l'intervallo di salto di $\tau$ in $s_0$, 
si ha che $\sigma$ ﺣ۷ costante su $I$ e dunque anche $u$ ﺣ۷ costante su $I$.
Dunque $\tilde u$ risulta essere continua anche nei punti in cui $\tau$ non lo ﺣ۷.
Inoltre, per il teorema~\ref{th:riparametrizzazione}, si ha che 
\[
  \ell_0^s(\tilde u) = s
\]
cioﺣ۷, come si dice, $\tilde u$ ﺣ۷ parametrizzata per lunghezza d'arco, in quanto 
la lunghezza del tratto di curva che va da $0$ a $s$ ﺣ۷ proprio $s$.

Siamo riusciti, senza usare le derivate, a definire la curva $\tilde u$ che percorre 
lo stesso tragitto di $u$ ma con una velocitﺣ  costante unitaria. 
Se $X=\RR^n$ e $u$ ﺣ۷ di classe $C^1$ con $u'(t)\neq 0$ per ogni $t\in[a,b]$,
allora si ha 
\[
  \sigma(t) = \int_a^t \abs{u'(t)}\, dt 
  \quad\implies\quad
  \sigma'(t) = \abs{u'(t)} 
\]
e dunque posto $\tilde u(s) = u(\tau(s))$ con $\tau(s) = \sigma^{-1}(s)$ si ha
\[
 \abs{\tilde u'(s)}
 = \abs{\frac{d}{ds} u(\sigma^{-1}(s))}
 = \frac{\abs{u'(\tau(s))}}{\sigma'(\sigma^{-1}(s))}
 = \frac{\abs{u'(\tau(s))}}{\abs{u'(\tau(s))}} = 1.
\]

\mynote{
Sebbene la derivata di una curva a valori in uno spazio metrico $X$ non possa essere definita
se $X$ non ha una struttura di spazio vettoriale, 
le osservazioni precedenti mostrano che puﺣﺎ avere senso definire il modulo 
della derivata, cioﺣ۷ la velocitﺣ , di una curva $u\colon [a,b] \to X$,
che viene chiamato \emph{derivata metrica} di $u$ e si indica con $\abs{u'}$:
\[
 \abs{u'(t_0)} \defeq \lim_{t\to t_0} \frac{d(u(t),u(t_0))}{t-t_0}.
\]
}

\begin{definition}[equivalenza di curve]
Due curve continue rettificabili $u \colon [a,b] \to X$ e $v \colon [A,B] \to X$ 
si dicono essere \emph{equivalenti} se le loro riparametrizzazioni per 
lunghezza d'arco coincidono.
\end{definition}

\section{Ascoli-Arzelﺣ }

\begin{theorem}[Ascoli-Arzelﺣ ]
Siano $\vec u_n \colon [a,b] \to \RR$ funzioni tali che 
    \begin{enumerate}
    \item le funzioni $\vec u_n$ sono uniformemente limitate, cioﺣ۷ esiste
    $M>0$ tale che per ogni $n$ e per ogni $t\in [a,b]$ si ha 
    $\abs{\vec u_n(t)} \le M$;
    \item le funzioni $\vec u_n$ sono equicontinue, cioﺣ۷ per ogni $\eps>0$ esiste 
    $\delta>0$ tale che per ogni $n$ e per ogni $t_1,t_2 \in [a,b]$ si ha 
    \[
    \abs{t_1-t_2}<\delta \implies 
    \abs{\vec u_n(t_1) - \vec u_n(t_2)} < \eps.
    \]
    \end{enumerate} 
Allora esiste una sottosuccessione $\vec u_{n_k}$ che converge uniformemente
ad una funzione continua $\vec u \colon [a,b] \to \RR$.
\end{theorem}   

\begin{proof}
\emph{Claim: per ogni $\eps>0$ esiste $n_k\in \NN$ successione strettamente 
crescente di indici, 
tale che $\Abs{\vec u_{n_k}(t) - \vec u_{n_j}(t)}_\infty < \eps$ per ogni $k,j\in \NN$.}
Dato $\delta>0$ scegliamo $N\in \NN$ tale che $N \delta < b-a$ e 
consideriamo $a=x_0 < x_1 < x_2 < \dots < x_N=b$ con $x_{i+1}-x_i < \delta$.
La successione $(u_n(x_1), u_n(x_2), \dots, u_n(x_N))\in \RR^N$ 
ﺣ۷ limitata dunque, per il teorema di Bolzano-Weierstrass,
ammette una estratta $(u_{n_k}(x_1),\dots,u_{n_k}(x_N))$ convergente in $\RR^N$.
Tale successione ﺣ۷ quindi di Cauchy in $\RR^N$. 
Dunque 
dato $\eps>0$ se $k,j$ sono sufficientemente grandi e per ogni $n=1,\dots,N$ si ha
\[
  \abs{u_{n_k}(x_n) - u_{n_j}(x_n)} < \eps.
\]
A meno di eliminare i primi indici della successione possiamo supporre che
questa disuguaglianza valga per ogni $k,j\in \NN$.

Ora, per l'equicontinuitﺣ , per ogni $t\in [a,b]$ esiste $x_n$ tale che
$\abs{t - x_n} < \delta$ e dunque
\begin{align*}
  \abs{u_{n_k}(t) - u_{n_j}(t)} 
  &\le \abs{u_{n_k}(t) - u_{n_k}(x_n)} + \abs{u_{n_k}(x_n) - u_{n_j}(x_n)} + \abs{u_{n_j}(x_n) - u_{n_j}(t)} \\
  &\le \eps + \eps + \eps = 3 \eps.
\end{align*}
Questo conclude la dimostrazione del claim.

\emph{Conclusione.}
Trovo $n_k^1$ tale che $\Abs{u_{n_k^1} - u_{n_j^1}}_\infty < 1$ per ogni $k,j$.
Poi trovo $n_k^2$ sottosuccessione estratta di $n_k^1$ tale che
$\Abs{u_{n_k^2} - u_{n_j^2}}_\infty < \frac 1 2$ e, 
procedendo in questo modo per induzione,
\[
  \Abs{u_{n_k^m} - u_{n_j^m}}_\infty < \frac 1 m \qquad \forall k,j \in \NN.
\]
La successione diagonale $u_{n_k^k}$ ha la proprietﺣ  che
\[
  \Abs{u_{n_k^k} - u_{n_j^j}} \le \frac 1 k \qquad \forall j\ge k.
\]
Quindi $u_{n_k^k}$ ﺣ۷ di Cauchy in $C^0([a,b],\RR)$ rispetto alla distanza 
uniforme e, per la completezza di tale spazio, converge uniformemente 
ad una funzione continua $u \colon [a,b] \to \RR$.
\end{proof}

La stessa dimostrazione permette di dimostrare il seguente risultato 
piﺣﺗ astratto.

\begin{theorem}[Ascoli-Arzelﺣ  piﺣﺗ astratto]
Sia $X$ uno spazio metrico compatto, $Y$ uno spazio metrico. 
Sia $u_n\colon X \to Y$ una successione di funzioni tali che 
\begin{enumerate}
    \item per ogni $x\in X$ la successione $u_n(x)$ ﺣ۷ relativamente compatta 
    in $Y$ (cioﺣ۷ ammette una estratta convergente);
    \item le funzioni $u_n$ sono equicontinue.
\end{enumerate}
Allora esiste una sottosuccessione $u_{n_k}$ che converge uniformemente
ad una funzione continua $u\colon X \to Y$.
\end{theorem}

Come applicazione del teorema di Ascoli-Arzelﺣ 
proponiamo il seguente importante risultato.

\begin{theorem}[esistenza delle geodetiche]
    Sia $X$ uno spazio metrico completo, proprio 
    (cioﺣ۷ ogni chiuso limitato ﺣ۷ compatto), 
    e sia $G(x,y)$ l'insieme di tutte le curve continue di lunghezza finita 
    $u\colon [0,1] \to X$
    tali che $u(0) = x$ e $u(1) = y$.
    Allora, se $G(x,y)$ non ﺣ۷ vuoto, il funzionale \emph{lunghezza}
    $\ell\colon G \to \RR$ ammette un minimo.
\end{theorem}
\begin{proof}
    Sia $L = \inf\ell(G)$ e sia 
    $u_n\in G(x,y)$ una successione minimizzante,
    cioﺣ۷ tale che $\ell(u_n) \to L$.
    Posto $L_n = \ell(u_n)$ posso considerare la riparametrizzazione 
    di $u_n$ per lunghezza d'arco. 
    Questo mi dﺣ  una curva $\tilde u_n \colon [0,L_n] \to X$
    che ﺣ۷ $1$-lipschitziana in quanto 
    $\ell(\tilde u_n, s,t) = t-s$.
    Le curve $v_n \colon [0,1] \to X$ definite da
    \[
        v_n(t) = \tilde u_n (L_n t)
    \]
    saranno invece $L_n$-lipschitziane. 
    Per il teorema~\ref{th:riparametrizzazione} si ha $\ell(v_n) = \ell(\tilde u_n) = \ell(u_n)$
    dunque $v_n\in G(x,y)$ e $\ell(v_n) = L_n$.
    Poichﺣ۸ $L_n\to L<+\infty$ esiste $M = \max L_n < +\infty$ 
    e le curve $v_n$ sono tutte $M$-lipschitziane e dunque 
    sono equicontinue. 
    Tutte queste curve sono contenute nella chiusura della 
    palla di centro $x$ e raggio $M$, che ﺣ۷ compatta
    perchﺣ۸ abbiamo supposto $X$ proprio.
    Dunque, per il teorema di Ascoli-Arzelﺣ , esiste una estratta 
    $v_{n_k}$ che converge uniformemente ad una curva continua
    $v\colon [0,1] \to X$.
    La curva $v$ ﺣ۷ $M$-lipschitziana in quanto limite uniforme di 
    $M$-lipschitziane, dunque ﺣ۷ di lunghezza finita e 
    appartiene a $G(x,y)$.
    Infine, per la semicontinuitﺣ  inferiore della lunghezza
    \[
        \ell(v) \le \liminf_{k\to \infty} \ell(v_{n_k}) = L,
    \]
    dunque $v$ ﺣ۷ una curva di lunghezza minima.
\end{proof}
